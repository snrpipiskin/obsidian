```go
package main

import (
	"fmt"
	"sync"
)

func main() {
	m := make(map[int]int)
	var wg sync.WaitGroup

	for i := 0; i < 1000; i++ {
		wg.Add(1)
		go func(n int) {
			defer wg.Done()
			for j := 0; j < 100; j++ {
				m[n*100+j] = n
			}
		}(i)
	}

	wg.Wait()
	fmt.Printf("Result: %d items\n", len(m))
}
```

---

fatal error: concurrent map writes (из-за конкурентной записи в мапу). Чтобы избежать ошибку можно использовать Mutex.
