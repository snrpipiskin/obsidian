```go
package main

import "fmt"

func modSlice(s []int) {
	for k, v := range s {
		s[k] = v * 2
	}
	s = append(s, 10)
}

func main() {
	s := []int{1, 2, 3}
	modSlice(s)
	fmt.Println(s)
}
```
