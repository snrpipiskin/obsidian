package main

import "fmt"

func mod(a []int) {
	for i, _ := range a {
		a[i] = 5
	}

	fmt.Println(a)
}

func main() {
	s1 := []int{1, 2, 3, 4}
	mod(s1)
	fmt.Println(s1)
}
