```go
package main

import (
	"encoding/json"
	"fmt"
)

func main() {
	type item struct {
		Name   string `json:"name"`
		number int64  `json:"number"`
	}

	items := []item{
		{
			Name:   "hello1",
			number: 1,
		},
		{
			Name:   "hello2",
			number: 2,
		},
		{
			Name:   "hello3",
			number: 3,
		},
	}

	for _, it := range items {
		it.Name = "request"
	}

	data, _ := json.Marshal(items)
	fmt.Println(string(data))
}
```
